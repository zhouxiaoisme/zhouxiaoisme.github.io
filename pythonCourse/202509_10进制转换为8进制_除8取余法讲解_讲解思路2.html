编程实现方法

Python实现

使用内置函数oct()可以快速将十进制转换为八进制：

# 使用oct()函数
decimal_num = 100
octal_num = oct(decimal_num)
print("八进制:", octal_num) # 输出: 0o144

# 手动实现除8取余法
def decimal_to_octal(num):
    remainders = []
    while num > 0:
        remainder = num % 8
        num = num // 8
        remainders.append(str(remainder))
    return ''.join(remainders[::-1])

C语言实现

使用printf函数的%o格式符可以输出八进制数：

#include <stdio.h>

int main() {
    int decimal_num = 100;
    // 使用%o格式符输出八进制
    printf("八进制: %o\n", decimal_num); // 输出: 144

    // 手动实现除8取余法
    int num = decimal_num;
    int remainders[20];
    int i = 0;
    while (num > 0) {
        remainders[i] = num % 8;
        num = num / 8;
        i++;
    }
    printf("手动计算八进制: ");
    for (int j = i-1; j >= 0; j--) {
        printf("%d", remainders[j]);
    }
    return 0;
}
